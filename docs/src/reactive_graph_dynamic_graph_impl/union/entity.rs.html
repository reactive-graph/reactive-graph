1use async_graphql::dynamic::*;
2use convert_case::Case::Pascal;
3use convert_case::Casing;
4
5use crate::DynamicGraphTypeDefinition;
6use reactive_graph_dynamic_graph_api::SchemaBuilderContext;
7
8pub const UNION_ALL_ENTITIES: &str = "AllEntities";
9pub const UNION_NAMESPACE_ENTITIES_SUFFIX: &str = "Entities";
10
11pub fn namespace_entities_union_type_name(namespace: &str) -> String {
12    format!("{}{}", namespace.to_case(Pascal), UNION_NAMESPACE_ENTITIES_SUFFIX)
13}
14
15pub fn get_namespace_entities_union(schema: SchemaBuilder, context: &SchemaBuilderContext, namespace: &String) -> SchemaBuilder {
16    let type_name = namespace_entities_union_type_name(namespace);
17    let mut union = Union::new(type_name).description(format!("Any entity of the namespace {}", namespace));
18    for entity_tys in context.entity_type_manager.get_types_by_namespace(namespace) {
19        let dy_ty = DynamicGraphTypeDefinition::from(&entity_tys);
20        union = union.possible_type(dy_ty.to_string());
21    }
22    schema.register(union)
23}
24
25pub fn get_all_entities_union(schema: SchemaBuilder, context: &SchemaBuilderContext) -> SchemaBuilder {
26    let mut union = Union::new(UNION_ALL_ENTITIES).description("Any entity.");
27    for entity_ty in context.entity_type_manager.get_type_ids() {
28        let dy_ty = DynamicGraphTypeDefinition::from(&entity_ty);
29        union = union.possible_type(dy_ty.to_string());
30    }
31    schema.register(union)
32}
