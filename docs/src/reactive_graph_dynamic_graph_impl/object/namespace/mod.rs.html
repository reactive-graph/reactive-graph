29pub fn namespace_query(context: SchemaBuilderContext, namespace: &String) -> Option<Object> {
30    let components = context.component_manager.get_by_namespace(namespace);
31    let entity_types = context.entity_type_manager.get_by_namespace(namespace);
32    let relation_types = context.relation_type_manager.get_by_namespace(namespace);
33    let flow_types = context.flow_type_manager.get_by_namespace(namespace);
34    if components.is_empty() && entity_types.is_empty() && relation_types.is_empty() && flow_types.is_empty() {
35        warn!("Skip empty query namespace {}", &namespace);
36        return None;
37    }
38    let type_name = namespace_type_name(namespace);
39    let namespace_field_value = namespace.clone();
40
41    let mut namespace =
42        Object::new(&type_name).description(format!("Queries for components, entities and relations on the namespace {}", &namespace.to_case(Pascal)));
43
44    for component in components.iter().sorted_by(sort_by_key) {
45        namespace = namespace.field(component_query_field(component.value()));
46        namespace = namespace.field(json_schema_field(&component.ty, component.json_schema()));
47    }
48
49    for entity_type in entity_types.iter().sorted_by(sort_by_key) {
50        let entity_type = entity_type.value();
51        namespace = namespace.field(entity_query_field(entity_type));
52        namespace = namespace.field(json_schema_field(&entity_type.ty, entity_type.json_schema()));
53    }
54    for relation_type in relation_types.iter().sorted_by(sort_by_key) {
55        namespace = namespace.field(relation_query_field(relation_type.value()));
56        namespace = namespace.field(json_schema_field(&relation_type.ty, relation_type.json_schema()));
57    }
58
59    for flow_type in flow_types.iter().sorted_by(sort_by_key) {
60        namespace = namespace.field(flow_query_field(flow_type.value()));
61        if let Some(entity_type) = entity_types.get(&flow_type.wrapper_type()) {
62            if let Ok(json_schema) = flow_type.json_schema(entity_type.value()) {
63                namespace = namespace.field(json_schema_field(&flow_type.ty, json_schema));
64            }
65        }
66    }
67
68    namespace = namespace.field(metrics_field(Some(namespace_field_value)));
69    Some(namespace)
70}

72pub fn namespace_mutation(context: SchemaBuilderContext, namespace: &String) -> Option<Object> {
73    let entity_types = context.entity_type_manager.get_by_namespace(namespace);
74    let relation_types = context.relation_type_manager.get_by_namespace(namespace);
75    let flow_types = context.flow_type_manager.get_by_namespace(namespace);
76    if entity_types.is_empty() && relation_types.is_empty() && flow_types.is_empty() {
77        warn!("Skip empty mutation namespace {}", &namespace);
78        return None;
79    }
80    let type_name = namespace_mutation_type_name(namespace);
81    let namespace_field_value = namespace.clone();
82
83    let mut namespace = Object::new(&type_name).description(format!("Mutations for entities and relations on the namespace {}", &namespace.to_case(Pascal)));
84
85    let mut contains_field = false;
86
87    for entity_type in entity_types.iter().sorted_by(|a, b| Ord::cmp(&a.key(), &b.key())) {
88        if let Some(field) = entity_creation_field(entity_type.value()) {
89            namespace = namespace.field(field);
90            contains_field = true;
91        }
92        if let Some(field) = entity_mutation_field(entity_type.value()) {
93            namespace = namespace.field(field);
94            contains_field = true;
95        }
96        // if let Some(field) = entity_export_field(entity_type.key()) {
97        //     namespace = namespace.field(field);
98        //     contains_field = true;
99        // }
100    }
101
102    for relation_type in relation_types.iter().sorted_by(|a, b| Ord::cmp(&a.key(), &b.key())) {
103        if let Some(field) = relation_creation_field(relation_type.value()) {
104            namespace = namespace.field(field);
105            contains_field = true;
106        }
107        if let Some(field) = relation_mutation_field(relation_type.value()) {
108            namespace = namespace.field(field);
109            contains_field = true;
110        }
111    }
112
113    for flow_type in flow_types.iter().sorted_by(|a, b| Ord::cmp(&a.key(), &b.key())) {
114        if let Some(field) = flow_creation_field(flow_type.value()) {
115            namespace = namespace.field(field);
116            contains_field = true;
117        }
118        if let Some(field) = flow_mutation_field(flow_type.value()) {
119            namespace = namespace.field(field);
120            contains_field = true;
121        }
122    }
123
124    namespace = namespace.field(metrics_field(Some(namespace_field_value)));
125
126    if contains_field { Some(namespace) } else { None }
127}
