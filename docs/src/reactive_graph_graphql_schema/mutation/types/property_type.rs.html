1use async_graphql::*;

5use reactive_graph_graph::PropertyType;

8use crate::query::GraphQLDataType;

11use crate::query::GraphQLSocketType;

13#[derive(Serialize, Deserialize, Clone, Debug, InputObject)]
14#[graphql(name = "PropertyTypeDefinition")]
15pub struct PropertyTypeDefinition {
16    /// The name of the property
17    pub name: String,
18
19    /// The description of the property
20    pub description: String,
21
22    /// The data type of the property
23    pub data_type: GraphQLDataType,
24
25    /// Specifies which type of socket
26    #[serde(default = "GraphQLSocketType::none")]
27    pub socket_type: GraphQLSocketType,
28
29    /// The property is mutable or immutable
30    #[serde(default = "GraphQLMutability::mutable")]
31    pub mutability: GraphQLMutability,
32
33    /// Property specific extensions
34    #[serde(default = "Vec::new")]
35    pub extensions: Vec<GraphQLExtension>,
36}
37
38impl From<PropertyTypeDefinition> for PropertyType {
39    fn from(property_type: PropertyTypeDefinition) -> Self {
40        PropertyType {
41            name: property_type.name,
42            description: property_type.description,
43            data_type: property_type.data_type.into(),
44            socket_type: property_type.socket_type.into(),
45            mutability: property_type.mutability.into(),
46            extensions: property_type.extensions.iter().map(|extension| extension.clone().into()).collect(),
47        }
48    }
49}
50
51#[derive(Default)]
52pub struct PropertyTypeDefinitions(pub Vec<PropertyTypeDefinition>);
53
54impl PropertyTypeDefinitions {
55    pub fn new(tys: Vec<PropertyTypeDefinition>) -> Self {
56        Self(tys)
57    }
58}
59
60impl From<PropertyTypeDefinitions> for PropertyTypes {
61    fn from(tys: PropertyTypeDefinitions) -> Self {
62        tys.0.into_iter().map(|ty| ty.into()).collect()
63    }
