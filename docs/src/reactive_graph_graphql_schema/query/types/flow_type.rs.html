24pub struct GraphQLFlowType {
25    flow_type: FlowType,
26}
27
28/// Flow types are templates for flow instances.
29#[Object(name = "FlowType")]
30impl GraphQLFlowType {
31    /// The entity type of the flow type.
32    #[graphql(name = "type")]
33    async fn entity_type(&self, context: &Context<'_>) -> Option<GraphQLEntityType> {
34        if let Ok(entity_type_manager) = context.data::<Arc<dyn EntityTypeManager + Send + Sync>>() {
35            return entity_type_manager.get(&self.flow_type.wrapper_entity_instance.ty).map(|e| e.into());
36        }
37        None
38    }
39
40    /// The namespace the flow type belongs to.
41    async fn namespace(&self) -> String {
42        self.flow_type.namespace()
43    }
44
45    /// The name of the flow type.
46    ///
47    /// The name is the unique identifier for flow types of the same namespace.
48    async fn name(&self) -> String {
49        self.flow_type.type_name()
50    }
51
52    /// Textual description of the flow type.
53    async fn description(&self) -> String {
54        self.flow_type.description.clone()
55    }
56
57    /// The wrapper entity instance.
58    async fn wrapper_entity_instance(&self) -> GraphQLEntityInstance {
59        let entity_instance = self.flow_type.wrapper_entity_instance.clone();
60        let reactive_entity = ReactiveEntity::from(entity_instance); // temporary, non-registered
61        GraphQLEntityInstance::from(reactive_entity)
62    }
63
64    /// The entity instances contained by the flow type
65    async fn entity_instances(&self) -> Vec<GraphQLEntityInstance> {
66        self.flow_type
67            .entity_instances
68            .iter()
69            .map(|entity_instance| {
70                // Construct a temporary instance, do not register!
71                let entity_instance: GraphQLEntityInstance = ReactiveEntity::from(entity_instance.clone()).into();
72                entity_instance
73            })
74            .collect()
75    }
76
77    /// The count of entity instances.
78    async fn count_entity_instances(&self) -> usize {
79        self.flow_type.entity_instances.len()
80    }
81
82    /// The relation instances contained by the flow type
83    async fn relation_instances(&self) -> Vec<GraphQLRelationInstance> {
84        self.flow_type
85            .relation_instances
86            .iter()
87            .filter_map(|relation_instance| {
88                // Construct temporary entity instances and a temporary relation instance, do not register!
89                let outbound = if relation_instance.outbound_id == self.flow_type.wrapper_entity_instance.id {
90                    ReactiveEntity::from(self.flow_type.wrapper_entity_instance.clone())
91                } else {
92                    self.flow_type
93                        .entity_instances
94                        .iter()
95                        .find(|e| e.id == relation_instance.outbound_id)
96                        .map(|e| ReactiveEntity::from(e.clone()))?
97                };
98                let inbound = if relation_instance.inbound_id == self.flow_type.wrapper_entity_instance.id {
99                    ReactiveEntity::from(self.flow_type.wrapper_entity_instance.clone())
100                } else {
101                    self.flow_type
102                        .entity_instances
103                        .iter()
104                        .find(|e| e.id == relation_instance.inbound_id)
105                        .map(|e| ReactiveEntity::from(e.clone()))?
106                };
107                let relation_instance = ReactiveRelation::new_from_instance(outbound, inbound, relation_instance.clone());
108                let relation_instance: GraphQLRelationInstance = relation_instance.into();
109                Some(relation_instance)
110            })
111            .collect()
112    }
113
114    /// The count of relation instances.
115    async fn count_relation_instances(&self) -> usize {
116        self.flow_type.relation_instances.len()
117    }
118
119    /// The properties of the flow type.
120    /// These are the properties of the wrapper entity instance.
121    async fn properties(
122        &self,
123        #[graphql(desc = "Filters by property name")] name: Option<String>,
124        #[graphql(desc = "Filters by property names")] names: Option<Vec<String>>,
125    ) -> Vec<GraphQLPropertyInstance> {
126        self.flow_type
127            .wrapper_entity_instance
128            .properties
129            .iter()
130            .filter(|property| name.is_none() || name.clone().unwrap().as_str() == property.key())
131            .filter(|property| names.is_none() || names.clone().unwrap().contains(property.key()))
132            .map(|property| {
133                // TODO: Resolve mutability using entity_type_manager and property_name
134                GraphQLPropertyInstance::new_entity_property(
135                    self.flow_type.wrapper_entity_instance.ty.clone(),
136                    property.key().clone(),
137                    property.value().clone(),
138                )
139            })
140            .collect()
141    }
142
143    /// The count of properties.
144    async fn count_properties(&self) -> usize {
145        self.flow_type.wrapper_entity_instance.properties.len()
146    }
147
148    /// The variables of the flow type.
149    async fn variables(&self, name: Option<String>) -> Vec<GraphQLPropertyType> {
150        if name.is_some() {
151            let name = name.unwrap();
152            return self
153                .flow_type
154                .variables
155                .to_vec()
156                .iter()
157                .filter(|property_type| property_type.name == name.clone())
158                .cloned()
159                .map(|property_type| property_type.into())
160                .collect();
161        }
162        self.flow_type
163            .variables
164            .iter()
165            .map(|property_type| property_type.value().clone().into())
166            .collect()
167    }
168
169    /// The count of variables.
170    async fn count_variables(&self) -> usize {
171        self.flow_type.variables.len()
172    }
173
174    /// The extensions which are defined by the flow type.
175    async fn extensions(&self, #[graphql(name = "type")] extension_ty: Option<ExtensionTypeIdDefinition>) -> Vec<GraphQLExtension> {
176        if let Some(extension_ty) = extension_ty {
177            let extension_ty = extension_ty.into();
178            return self
179                .flow_type
180                .extensions
181                // .to_vec()
182                .iter()
183                .filter(|extension| extension.ty == extension_ty)
184                .map(|extension| extension.value().clone().into())
185                .collect();
186        }
187        self.flow_type.extensions.iter().map(|extension| extension.value().into()).collect()
188    }
189
190    /// The count of extensions.
191    async fn count_extensions(&self) -> usize {
192        self.flow_type.extensions.len()
193    }
194
195    /// Returns true, if the relation type is valid.
196    ///
197    /// This means all components exists and the outbound and inbound entity types are valid.
198    async fn is_valid(&self, context: &Context<'_>) -> bool {
199        match context.data::<Arc<dyn FlowTypeManager + Send + Sync>>() {
200            Ok(flow_type_manager) => flow_type_manager.validate(&self.flow_type.ty),
201            Err(_) => false,
202        }
203    }
204
205    /// Returns the JSON schema of the flow type.
206    async fn json_schema(&self, context: &Context<'_>) -> Result<Value> {
207        let entity_type_manager = context.data::<Arc<dyn EntityTypeManager + Send + Sync>>()?;
208        let entity_ty = self.flow_type.wrapper_type();
209        let entity_type = entity_type_manager
210            .get(&entity_ty)
211            .ok_or(Error::new(format!("Missing wrapper entity type {entity_ty}")))?;
212        match self.flow_type.json_schema(&entity_type) {
213            Ok(schema) => Ok(schema.to_value()),
214            Err(e) => Err(e.into()),
215        }
216    }
217
218    /// Returns the JSON schema identifier ($id) of the flow type.
219    async fn json_schema_id(&self) -> String {
220        self.flow_type.json_schema_id().to_string()
221    }
222}

225    fn from(flow_type: FlowType) -> Self {
226        GraphQLFlowType { flow_type }
227    }
