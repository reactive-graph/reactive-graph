25    }
26
27    /// Returns the JSON schema for relation types.
28    async fn relations(&self, _context: &Context<'_>) -> Result<Value> {
29        Ok(schema_relation_types().to_value())
30    }
31
32    /// Returns the JSON schema for flow types.
