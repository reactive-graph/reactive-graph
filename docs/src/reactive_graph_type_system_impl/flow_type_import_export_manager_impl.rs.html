23#[derive(Component)]
24pub struct FlowTypeImportExportManagerImpl {
25    flow_type_manager: Arc<dyn FlowTypeManager + Send + Sync>,
26}
27
28#[async_trait]
29#[component_alias]
30impl FlowTypeImportExportManager for FlowTypeImportExportManagerImpl {
31    async fn import(&self, path: &str) -> Result<FlowType, FlowTypeImportError> {
32        let path = Path::new(path);
33        let file = File::open(path)?;
34        let mut reader = BufReader::new(file);
35        let mut content = String::new();
36        reader.read_to_string(&mut content)?;
37        let flow_type = match path.extension().and_then(|ext| ext.to_str()) {
38            Some("json") => serde_json::from_str::<FlowType>(&content).map_err(|e| DeserializationError::Json(e).into()),
39            #[cfg(feature = "json5")]
40            Some("json5") => json5::from_str::<FlowType>(&content).map_err(|e| DeserializationError::Json5(e).into()),
41            #[cfg(feature = "toml")]
42            Some("toml") => toml::from_str::<FlowType>(&content).map_err(|e| DeserializationError::Toml(e).into()),
43            Some(ext) => Err(FlowTypeImportError::UnsupportedFormat(ext.to_string())),
44            None => Err(FlowTypeImportError::UnsupportedFormat(Default::default())),
45        }?;
46        self.flow_type_manager.register(flow_type).map_err(FlowTypeImportError::RegistrationError)
47    }
48
49    async fn export(&self, ty: &FlowTypeId, path: &str) -> Result<(), FlowTypeExportError> {
50        let Some(flow_type) = self.flow_type_manager.get(ty) else {
51            return Err(FlowTypeExportError::FlowTypeNotFound(ty.clone()));
52        };
53        let path = Path::new(path);
54        match path.extension().and_then(|ext| ext.to_str()) {
55            Some("json") => match File::create(path) {
56                Ok(file) => serde_json::to_writer_pretty(&file, &flow_type).map_err(|e| SerializationError::Json(e).into()),
57                Err(e) => Err(FlowTypeExportError::Io(e)),
58            },
59            #[cfg(feature = "json5")]
60            Some("json5") => match File::create(path) {
61                Ok(mut file) => {
62                    let content = json5::to_string(&flow_type).map_err(|e| FlowTypeExportError::Serialization(SerializationError::Json5(e)))?;
63                    file.write_all(content.as_bytes()).map_err(FlowTypeExportError::Io)
64                }
65                Err(e) => Err(FlowTypeExportError::Io(e)),
66            },
67            #[cfg(feature = "toml")]
68            Some("toml") => match File::create(path) {
69                Ok(mut file) => {
70                    let content = toml::to_string_pretty(&flow_type).map_err(|e| FlowTypeExportError::Serialization(SerializationError::Toml(e)))?;
71                    file.write_all(content.as_bytes()).map_err(FlowTypeExportError::Io)
72                }
73                Err(e) => Err(FlowTypeExportError::Io(e)),
74            },
75            Some(ext) => Err(FlowTypeExportError::UnsupportedFormat(ext.to_string())),
76            None => Err(FlowTypeExportError::UnsupportedFormat(Default::default())),
77        }
78    }
79}
