68    fn register(&self, flow_type: FlowType) -> Result<FlowType, FlowTypeRegistrationError> {
69        let flow_ty = flow_type.ty.clone();
70        if self.has(&flow_ty) {
71            return Err(FlowTypeRegistrationError::FlowTypeAlreadyExists(flow_ty));
72        }
73        // Check that the entity types of every declared entity instance exists
74        for entity_ty in flow_type.uses_entity_types() {
75            if !self.entity_type_manager.has(&entity_ty) {
76                warn!("Flow type {flow_ty} not fully initialized: No entity type {entity_ty}");
77            }
78        }
79        // Check that the relation type of every declared relation instance exists
80        for relation_ty in flow_type.uses_relation_types() {
81            if !self.relation_type_manager.has(&relation_ty) {
82                warn!("Flow type {flow_ty} not fully initialized: No relation type named {relation_ty}");
83            }
84        }
85        // TODO: Check that entity instances referenced by a relation instance exists
86        // TODO: Check that relation instances outbound entity has correct entity_type
87        // TODO: Check that relation instances inbound entity has correct entity_type
88        self.flow_types.push(flow_type.clone());
89        debug!("Registered flow type {flow_ty}");
90        self.event_manager.emit_event(TypeSystemEvent::FlowTypeCreated(flow_ty));
91        Ok(flow_type)
92    }

142    fn create_flow_type(
143        &self,
144        ty: &FlowTypeId,
145        description: &str,
146        wrapper_entity_instance: EntityInstance,
147        entity_instances: EntityInstances,
148        relation_instances: RelationInstances,
149        variables: PropertyTypes,
150        extensions: Extensions,
151    ) -> Result<FlowType, FlowTypeCreationError> {
152        let flow_type = FlowType::builder()
153            .ty(ty)
154            .description(description)
155            .wrapper_entity_instance(wrapper_entity_instance)
156            .entity_instances(entity_instances)
157            .relation_instances(relation_instances)
158            .variables(variables)
159            .extensions(extensions)
160            .build();
161        self.register(flow_type).map_err(FlowTypeCreationError::RegistrationError)
162    }

164    fn update_description(&self, ty: &FlowTypeId, description: &str) -> Result<FlowType, FlowTypeUpdateError> {
165        if !self.has(ty) {
166            return Err(FlowTypeUpdateError::FlowTypeDoesNotExist(ty.clone()));
167        }
168        for mut flow_type in self.flow_types.iter_mut() {
169            if &flow_type.ty == ty {
170                flow_type.description = description.to_string();
171                // TODO: Notify about changed flow_type
172            }
173        }
174        self.get(ty).ok_or(FlowTypeUpdateError::FlowTypeDoesNotExist(ty.clone()))
175    }

260    fn validate(&self, ty: &FlowTypeId) -> bool {
261        if let Some(flow_type) = self.get(ty) {
262            return flow_type
263                .entity_instances
264                .iter()
265                .all(|entity_instance| self.entity_type_manager.validate(&entity_instance.ty))
266                && flow_type
267                    .relation_instances
268                    .iter()
269                    .all(|relation_instance| self.relation_type_manager.validate(&relation_instance.relation_type_id()));
270        }
271        false
272    }
