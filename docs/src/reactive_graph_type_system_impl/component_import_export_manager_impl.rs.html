27#[async_trait]
28#[component_alias]
29impl ComponentImportExportManager for ComponentImportExportManagerImpl {
30    async fn import(&self, path: &str) -> Result<reactive_graph_graph::Component, ComponentImportError> {
31        let path = Path::new(path);
32        let file = File::open(path)?;
33        let mut reader = BufReader::new(file);
34        let mut content = String::new();
35        reader.read_to_string(&mut content)?;
36        let component = match path.extension().and_then(|ext| ext.to_str()) {
37            Some("json") => serde_json::from_str::<reactive_graph_graph::Component>(&content).map_err(|e| DeserializationError::Json(e).into()),
38            #[cfg(feature = "json5")]
39            Some("json5") => json5::from_str::<reactive_graph_graph::Component>(&content).map_err(|e| DeserializationError::Json5(e).into()),
40            #[cfg(feature = "toml")]
41            Some("toml") => toml::from_str::<reactive_graph_graph::Component>(&content).map_err(|e| DeserializationError::Toml(e).into()),
42            Some(ext) => Err(ComponentImportError::UnsupportedFormat(ext.to_string())),
43            None => Err(ComponentImportError::UnsupportedFormat(Default::default())),
44        }?;
45        self.component_manager.register(component).map_err(ComponentImportError::RegistrationError)
46    }
47
48    async fn export(&self, ty: &ComponentTypeId, path: &str) -> Result<(), ComponentExportError> {
49        let Some(component) = self.component_manager.get(ty) else {
50            return Err(ComponentExportError::ComponentNotFound(ty.clone()));
51        };
52        let path = Path::new(path);
53        match path.extension().and_then(|ext| ext.to_str()) {
54            Some("json") => match File::create(path) {
55                Ok(file) => serde_json::to_writer_pretty(&file, &component).map_err(|e| SerializationError::Json(e).into()),
56                Err(e) => Err(ComponentExportError::Io(e)),
57            },
58            #[cfg(feature = "json5")]
59            Some("json5") => match File::create(path) {
60                Ok(mut file) => {
61                    let content = json5::to_string(&component).map_err(|e| ComponentExportError::Serialization(SerializationError::Json5(e)))?;
62                    file.write_all(content.as_bytes()).map_err(ComponentExportError::Io)
63                }
64                Err(e) => Err(ComponentExportError::Io(e)),
65            },
66            #[cfg(feature = "toml")]
67            Some("toml") => match File::create(path) {
68                Ok(mut file) => {
69                    let content = toml::to_string_pretty(&component).map_err(|e| ComponentExportError::Serialization(SerializationError::Toml(e)))?;
70                    file.write_all(content.as_bytes()).map_err(ComponentExportError::Io)
71                }
72                Err(e) => Err(ComponentExportError::Io(e)),
73            },
74            Some(ext) => Err(ComponentExportError::UnsupportedFormat(ext.to_string())),
75            None => Err(ComponentExportError::UnsupportedFormat(Default::default())),
76        }
77    }
78}

95    async fn test_export_import_component() {
96        reactive_graph_test_utils::init_logger();
97        let type_system = reactive_graph_di::get_container::<TypeSystemImpl>();
98        let component_manager = type_system.get_component_manager();
99        let component_import_export_manager = type_system.get_component_import_export_manager();
100
101        let component = component_manager.register(Component::default_test()).expect("Failed to create component");
102        let component_orig = component.clone();
103        let component_ty = component.ty.clone();
104
105        let mut path = env::temp_dir();
106        path.push(format!("{}__{}.json", component_ty.namespace(), component_ty.type_name()));
107        let path = path.into_os_string().into_string().unwrap();
108
109        component_import_export_manager
110            .export(&component_ty, path.as_str())
111            .await
112            .expect("Failed to export component");
113        assert!(component_manager.has(&component_ty), "Component should be registered!");
114        assert!(component_manager.delete(&component_ty), "Failed to delete component!");
115        assert!(!component_manager.has(&component_ty), "Component shouldn't be registered anymore!");
116        let _component = component_import_export_manager.import(path.as_str()).await.expect("Failed to import component");
117        assert!(component_manager.has(&component_ty), "Component not registered!");
118        assert_eq!(
119            component_orig,
120            component_manager.get(&component_ty).unwrap(),
121            "The imported component should match the constructed component!"
122        );
123    }
