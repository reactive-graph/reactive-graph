12pub fn dynamic_json_schema_for_entity_type(entity_type: EntityType) -> Schema {
13    let type_name = entity_type.type_name();
14    let schema_id = dynamic_schema_id(&entity_type.ty);
15    let mut properties = Map::new();
16    properties.insert(
17        "$id".to_string(),
18        json!({
19            "default": schema_id,
20            "description": "The schema identifier",
21            "type": "string"
22        }),
23    );
24    properties.insert(
25        "id".to_string(),
26        json!({
27            "description": format!("The unique identifier of an entity instance of entity type {}", type_name),
28            "type": "string",
29            "format": "uuid"
30        }),
31    );
32    for entry in entity_type.properties.iter() {
33        // TODO: sort
34        let property = entry.value();
35        let property_name = entry.key().clone();
36        if property_name == "$id" || property_name == "id" {
37            continue;
38        }
39        let mut json_schema_property = Map::new();
40        json_schema_property.insert("type".to_string(), JsonSchemaDataType::new(property.data_type).to_value());
41        if let Some(extension_json_schema_format) = property.get_extension(&EXTENSION_JSON_SCHEMA_FORMAT).map(|extension| extension.extension) {
42            json_schema_property.insert("format".to_string(), extension_json_schema_format);
43        }
44        if !property.description.is_empty() {
45            json_schema_property.insert("description".to_string(), json!(&property.description));
46        }
47        properties.insert(property_name, Value::Object(json_schema_property));
48    }
49    // TODO: Extract required from property extension
50    let mut required = entity_type.properties.names();
51    required.push("id".to_string());
52    required.sort();
53    json_schema!({
54        "$id": schema_id,
55        "type": "object",
56        "title": entity_type.type_name(),
57        "description": entity_type.description,
58        "properties": properties,
59        "required": required,
60    })
61}
