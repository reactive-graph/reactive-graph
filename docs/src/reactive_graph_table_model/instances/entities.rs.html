28#[derive(Clone, Debug, Tabled)]
29pub struct EntityInstance {
30    /// The id of the entity instance.
31    pub id: Uuid,
32
33    /// The type namespace.
34    pub namespace: String,
35
36    /// The type name.
37    pub name: String,
38
39    /// Textual description of the entity instance.
40    pub description: String,
41
42    /// The components.
43    #[tabled(display_with("Self::display_component_type_ids", self))]
44    pub components: Vec<ComponentTypeId>,
45
46    /// The property instances.
47    #[tabled(display_with("Self::display_property_instances", self))]
48    pub properties: Vec<PropertyInstance>,
49
50    /// The extensions.
51    #[tabled(display_with("Self::display_extensions", self))]
52    pub extensions: Vec<Extension>,
53
54    #[tabled(skip)]
55    inline_format: TableInlineFormat,
56}
57
58impl EntityInstance {
59    fn display_component_type_ids(&self) -> String {
60        match self.inline_format {
61            TableInlineFormat::Table => display_component_type_ids_inline_str(&self.components),
62            TableInlineFormat::Html => display_component_type_ids_html_inline(&self.components),
63        }
64    }
65
66    fn display_property_instances(&self) -> String {
67        match self.inline_format {
68            TableInlineFormat::Table => display_property_instances_inline_str(&self.properties),
69            TableInlineFormat::Html => display_property_instances_html_inline(&self.properties),
70        }
71    }
72
73    fn display_extensions(&self) -> String {
74        // println!("{:?}", &self.inline_format);
75        match self.inline_format {
76            TableInlineFormat::Table => display_extensions_inline_str(&self.extensions),
77            TableInlineFormat::Html => display_extensions_html_inline(&self.extensions),
78        }
79    }
80}
81
82impl TableInlineFormatSetter for EntityInstance {
83    fn set_table_inline_format(&mut self, table_inline_format: TableInlineFormat) {
84        self.inline_format = table_inline_format;
85    }
86}

88impl From<reactive_graph_graph::EntityInstance> for EntityInstance {
89    fn from(entity_instance: reactive_graph_graph::EntityInstance) -> Self {
90        EntityInstance {
91            id: entity_instance.id,
92            namespace: entity_instance.namespace(),
93            name: entity_instance.type_name(),
94            description: entity_instance.description,
95            components: ComponentTypeIds::from(entity_instance.components).0,
96            properties: PropertyInstances::from(entity_instance.properties).0,
97            extensions: Extensions::from(entity_instance.extensions).0,
98            inline_format: Default::default(),
99        }
100    }
101}

105impl TableOptions for EntityInstancesTableOptions {
106    fn options(table: &mut Table) -> &mut Table {
107        table.with(Style::extended()).with(
108            Modify::new(Segment::new(0.., 0..3))

116pub fn display_entity_instances_inline_str(entity_instances: &[EntityInstance]) -> String {
117    if entity_instances.is_empty() {
118        String::new()
119    } else {
120        display_entity_instances_inline(entity_instances).to_string()
121    }
122}
123
124pub fn display_entity_instances_inline(entity_instances: &[EntityInstance]) -> Table {
125    let entity_instances = entity_instances.to_vec();
126    Table::new(entity_instances)
127        .with(modern_inline())
128        .with(Modify::new(Columns::new(0..1)).with(Width::increase(22)))
129        // .with(Modify::new(Columns::new(1..2)).with(Width::increase(22)))
130        // .with(Modify::new(Columns::new(2..3)).with(Width::wrap(40)))
131        // .with(Modify::new(Columns::new(3..4)).with(Width::wrap(80)))
132        .to_owned()
133}
134
135pub fn display_entity_instances_html_inline(entity_instances: &[EntityInstance]) -> String {
136    let entity_instances = entity_instances.to_vec();
137    if entity_instances.is_empty() {
138        return String::new();
139    }
140    HtmlTable::with_header(Vec::<Vec<String>>::from(Table::builder(&entity_instances)))
141        .to_string()
142        .split_whitespace()
143        .collect()
144}
