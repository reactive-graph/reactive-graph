26#[injectable]
27#[async_trait]
28pub trait ComponentManager: Send + Sync + Lifecycle {
29    /// Registers the given component type.
30    fn register(&self, component: Component) -> Result<Component, ComponentRegistrationError>;
31
32    /// Returns all component types.
33    fn get_all(&self) -> Components;
34
35    /// Returns the ids of all component types.
36    fn get_type_ids(&self) -> ComponentTypeIds;
37
38    /// Returns all defined namespaces.
39    fn get_namespaces(&self) -> Namespaces;
40
41    /// Returns all components of the given namespace.
42    fn get_by_namespace(&self, namespace: &str) -> Components;
43
44    /// Returns all component types of the given namespace.
45    fn get_types_by_namespace(&self, namespace: &str) -> ComponentTypeIds;
46
47    /// Returns true, if a component with the given type exists.
48    fn has(&self, ty: &ComponentTypeId) -> bool;
49
50    /// Returns true, if a component with the given fully qualified name exists.
51    fn has_by_type(&self, namespace: &str, type_name: &str) -> bool;
52
53    /// Returns the component with the given name or empty.
54    fn get(&self, ty: &ComponentTypeId) -> Option<Component>;
55
56    /// Returns the component with the given fully qualified name or empty.
57    fn get_by_type(&self, namespace: &str, type_name: &str) -> Option<Component>;
58
59    /// Returns the components with the given types.
60    fn get_by_types(&self, tys: ComponentTypeIds) -> Components;
61
62    /// Returns all components whose type names matches the given search string.
63    fn find_by_type_name(&self, search: &str) -> Components;
64
65    /// Returns the count of components.
66    fn count(&self) -> usize;
67
68    /// Returns the count of components of the given namespace.
69    fn count_by_namespace(&self, namespace: &str) -> usize;
70
71    /// Creates a new component of the given type with the description, properties and extensions.
72    fn create_component(
73        &self,
74        ty: &ComponentTypeId,
75        description: &str,
76        properties: PropertyTypes,
77        extensions: Extensions,
78    ) -> Result<Component, ComponentCreationError>;
79
80    /// Replaces the component with the given name with the given component.
81    fn replace(&self, ty: &ComponentTypeId, component: Component);
82
83    /// Updates the description of the given component.
84    fn update_description(&self, ty: &ComponentTypeId, description: &str) -> Result<Component, ComponentUpdateError>;
85
86    /// Merges the given component_to_merge into an existing component with the same component type id.
87    fn merge(&self, component_to_merge: Component) -> Result<Component, ComponentMergeError>;
88
89    /// Adds a property to the given component.
90    fn add_property(&self, ty: &ComponentTypeId, property: PropertyType) -> Result<PropertyType, ComponentAddPropertyError>;
91
92    /// Updates the property with the given property_name.
93    /// It's possible to rename the property by using another name in the new property than the provided property_name.
94    fn update_property(&self, ty: &ComponentTypeId, property_name: &str, property: PropertyType) -> Result<PropertyType, ComponentUpdatePropertyError>;
95
96    /// Removes the property with the given property_name from the given component.
97    fn remove_property(&self, ty: &ComponentTypeId, property_name: &str) -> Result<PropertyType, ComponentRemovePropertyError>;
98
99    /// Adds an extension to the given component.
100    fn add_extension(&self, ty: &ComponentTypeId, extension: Extension) -> Result<ExtensionTypeId, ComponentAddExtensionError>;
101
102    /// Replaces the extension of the given component.
103    fn update_extension(
104        &self,
105        component_ty: &ComponentTypeId,
106        extension_ty: &ExtensionTypeId,
107        extension: Extension,
108    ) -> Result<Extension, ComponentUpdateExtensionError>;
109
110    /// Removes the extension with the given extension_name from the given component.
111    fn remove_extension(&self, component_ty: &ComponentTypeId, extension_ty: &ExtensionTypeId) -> Result<Extension, ComponentRemoveExtensionError>;
112
113    /// Deletes the component with the given name.
114    fn delete(&self, ty: &ComponentTypeId) -> bool;
115}
