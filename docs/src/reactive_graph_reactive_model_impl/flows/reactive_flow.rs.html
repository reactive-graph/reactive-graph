32pub struct ReactiveFlowInstance {
33    /// The id of the flow corresponds to the id of the wrapper entity instance.
34    pub id: Uuid,
35
36    /// The type definition of the entity type of the wrapper entity instance.
37    pub ty: EntityTypeId,
38
39    /// The name of the flow instance.
40    pub name: String,
41
42    /// Textual description of the flow instance.
43    pub description: String,
44
45    /// The flow contains entity instances. The entity instance may also
46    /// be contained in other flows.
47    pub entity_instances: RwLock<HashMap<Uuid, ReactiveEntity>>,
48
49    /// The flow contains relation instances. The relation instances may also
50    /// be contained in other flows.
51    pub relation_instances: RwLock<HashMap<RelationInstanceId, ReactiveRelation>>,
52
53    /// List of entities that has been added since creation of the flow.
54    pub entities_added: RwLock<Vec<Uuid>>,
55
56    /// List of entities that has been removed since creation of the flow.
57    pub entities_removed: RwLock<Vec<Uuid>>,
58
59    /// List of relations that has been added since creation of the flow.
60    pub relations_added: RwLock<Vec<RelationInstanceId>>,
61
62    /// List of relations that has been removed since creation of the flow.
63    pub relations_removed: RwLock<Vec<RelationInstanceId>>,
64}

66impl ReactiveFlowInstance {
67    pub fn new(wrapper_entity_instance: ReactiveEntity) -> Self {
68        let mut entity_instances = HashMap::new();
69        entity_instances.insert(wrapper_entity_instance.id, wrapper_entity_instance.clone());
70        ReactiveFlowInstance {
71            id: wrapper_entity_instance.id,
72            ty: wrapper_entity_instance.ty.clone(),
73            name: String::new(),
74            description: String::new(),
75            entity_instances: RwLock::new(entity_instances),
76            relation_instances: RwLock::new(HashMap::new()),
77            entities_added: RwLock::new(Vec::new()),
78            entities_removed: RwLock::new(Vec::new()),
79            relations_added: RwLock::new(Vec::new()),
80            relations_removed: RwLock::new(Vec::new()),
81        }
82    }
83
84    pub fn has_entity(&self, entity_instance: ReactiveEntity) -> bool {
85        self.entity_instances.read().unwrap().contains_key(&entity_instance.id)
86    }
87
88    pub fn has_entity_by_id(&self, id: Uuid) -> bool {
89        self.entity_instances.read().unwrap().contains_key(&id)
90    }
91
92    pub fn get_entity(&self, id: Uuid) -> Option<ReactiveEntity> {
93        let reader = self.entity_instances.read().unwrap();
94        reader.get(&id).cloned()
95    }
96
97    pub fn get_wrapper_entity_instance(&self) -> Option<ReactiveEntity> {
98        self.get_entity(self.id)
99    }
100
101    pub fn add_entity(&self, entity_instance: ReactiveEntity) {
102        let id = entity_instance.id;
103        if !self.has_entity_by_id(id) {
104            self.entity_instances.write().unwrap().insert(id, entity_instance);
105            self.entities_added.write().unwrap().push(id);
106            // self.entities_removed.write().unwrap().remove(entity_instance.id);
107        }
108    }
109
110    pub fn remove_entity(&self, id: Uuid) {
111        self.entity_instances.write().unwrap().remove(&id);
112        self.entities_removed.write().unwrap().push(id);
113    }
114
115    pub fn has_relation(&self, relation_instance: ReactiveRelation) -> bool {
116        self.relation_instances.read().unwrap().contains_key(&relation_instance.id())
117    }
118
119    pub fn has_relation_by_key(&self, relation_instance_id: &RelationInstanceId) -> bool {
120        self.relation_instances.read().unwrap().contains_key(relation_instance_id)
121    }
122
123    pub fn get_relation(&self, relation_instance_id: &RelationInstanceId) -> Option<ReactiveRelation> {
124        let reader = self.relation_instances.read().unwrap();
125        reader.get(relation_instance_id).cloned()
126    }
127
128    pub fn add_relation(&self, relation_instance: ReactiveRelation) {
129        let id = relation_instance.id();
130        if !self.has_relation_by_key(&id) {
131            self.relation_instances.write().unwrap().insert(id.clone(), relation_instance);
132            self.relations_added.write().unwrap().push(id);
133        }
134    }
135
136    pub fn remove_relation(&self, relation_instance_id: &RelationInstanceId) {
137        self.relation_instances.write().unwrap().remove(relation_instance_id);
138        self.relations_removed.write().unwrap().push(relation_instance_id.clone());
139    }
140
141    pub fn tick(&self) {
142        let reader = self.entity_instances.read().unwrap();
143        for (_, entity_instance) in reader.iter() {
144            entity_instance.tick();
145        }
146    }
147}

155impl TryFrom<FlowInstance> for ReactiveFlowInstance {
156    type Error = ReactiveFlowConstructionError;
157
158    fn try_from(flow_instance: FlowInstance) -> Result<Self, ReactiveFlowConstructionError> {
159        let flow_id = flow_instance.id;
160        let mut entity_instances = HashMap::new();
161        let mut wrapper = None;
162        for (id, entity_instance) in flow_instance.entity_instances {
163            let reactive_entity = ReactiveEntity::from(entity_instance);
164            entity_instances.insert(id, reactive_entity.clone());
165            if id == flow_id {
166                wrapper = Some(reactive_entity.clone());
167            }
168        }
169        if wrapper.is_none() {
170            return Err(ReactiveFlowConstructionError::MissingWrapperInstance);
171        }
172        let mut relation_instances = HashMap::new();
173        for (id, relation_instance) in flow_instance.relation_instances {
174            // let id = relation_instance.id();
175            let outbound = entity_instances.get(&relation_instance.outbound_id);
176            if outbound.is_none() {
177                // outbound entity missing
178                return Err(ReactiveFlowConstructionError::MissingOutboundEntityInstance(relation_instance.outbound_id));
179            }
180            let inbound = entity_instances.get(&relation_instance.inbound_id);
181            if inbound.is_none() {
182                // inbound entity missing
183                return Err(ReactiveFlowConstructionError::MissingInboundEntityInstance(relation_instance.inbound_id));
184            }
185            let outbound = outbound.unwrap().clone();
186            let inbound = inbound.unwrap().clone();
187            let reactive_relation = ReactiveRelation::new_from_instance(outbound, inbound, relation_instance.clone());
188            relation_instances.insert(id, reactive_relation);
189        }
190        Ok(ReactiveFlowInstance {
191            id: flow_id,
192            ty: flow_instance.ty,
193            name: flow_instance.name.clone(),
194            description: flow_instance.description.clone(),
195            entity_instances: RwLock::new(entity_instances),
196            relation_instances: RwLock::new(relation_instances),
197            // wrapper: wrapper.unwrap(),
198            entities_added: RwLock::new(Vec::new()),
199            entities_removed: RwLock::new(Vec::new()),
200            relations_added: RwLock::new(Vec::new()),
201            relations_removed: RwLock::new(Vec::new()),
202        })
203    }
204}

319impl TryFrom<ReactiveFlowInstance> for FlowInstance {
320    type Error = CreateFlowInstanceError;
321
322    fn try_from(reactive_flow: ReactiveFlowInstance) -> Result<Self, CreateFlowInstanceError> {
323        let wrapper = reactive_flow.get_entity(reactive_flow.id);
324        if wrapper.is_none() {
325            return Err(CreateFlowInstanceError::CantGetWrapperEntity(reactive_flow.id));
326        }
327        let wrapper = wrapper.unwrap();
328        let entity_instance: EntityInstance = wrapper.clone().into();
329        let mut flow_instance = FlowInstance::from(entity_instance);
330        flow_instance.description = wrapper.description.clone();
331        reactive_flow.entity_instances.read().unwrap().iter().for_each(|(_, reactive_entity)| {
332            if reactive_entity.id != reactive_flow.id {
333                let entity_instance: EntityInstance = reactive_entity.into();
334                flow_instance.entity_instances.push(entity_instance);
335            }
336        });
337        reactive_flow.relation_instances.read().unwrap().iter().for_each(|(_, reactive_instance)| {
338            let relation_instance = RelationInstance::from(reactive_instance);
339            flow_instance.relation_instances.push(relation_instance);
340        });
341        Ok(flow_instance)
342    }
343}
