18#[derive(Component)]
19pub struct EntityInstanceImportExportManagerImpl {
20    reactive_entity_manager: Arc<dyn ReactiveEntityManager + Send + Sync>,
21}
22
23#[async_trait]
24#[component_alias]
25impl EntityInstanceImportExportManager for EntityInstanceImportExportManagerImpl {
26    async fn import(&self, path: &str) -> Result<ReactiveEntity, EntityInstanceImportError> {
27        let file = File::open(path)?;
28        let reader = BufReader::new(file);
29        let entity_instance: EntityInstance = serde_json::from_reader(reader)?;
30        if self.reactive_entity_manager.has(entity_instance.id) {
31            return Err(EntityInstanceImportError::EntityAlreadyExists(entity_instance.id));
32        }
33        self.reactive_entity_manager
34            .create_reactive_instance(entity_instance)
35            .map_err(EntityInstanceImportError::ReactiveEntityCreationError)
36    }
37
38    async fn export(&self, id: Uuid, path: &str) -> Result<(), EntityInstanceExportError> {
39        let Some(entity_instance) = self.reactive_entity_manager.get(id) else {
40            return Err(EntityInstanceExportError::EntityNotFound(id));
41        };
42        let Ok(file) = File::create(path) else {
43            return Err(EntityInstanceExportError::Io(String::from(path)));
44        };
45        let entity_instance = EntityInstance::from(entity_instance);
46        serde_json::to_writer_pretty(&file, &entity_instance).map_err(EntityInstanceExportError::Serialization)
47    }
48}

86        let entity_type = entity_type_manager

88            .expect("Failed to register entity type");
89
90        let mut path = env::temp_dir();
91        path.push(format!("{}.json", entity_type.type_name()));
92        let path = path.into_os_string().into_string().unwrap();
93
94        let entity_instance = EntityInstance::default_from(&entity_type);
95
96        let reactive_entity = reactive_entity_manager
97            .create_reactive_instance(entity_instance.clone())
98            .expect("Failed to create reactive instance");
99        let uuid = reactive_entity.id;
100
101        entity_instance_import_export_manager
102            .export(uuid, &path)
103            .await
104            .expect("Failed to export entity instance");
105        assert!(reactive_entity_manager.has(uuid), "Missing reactive entity with uuid!");
106        assert!(reactive_entity_manager.delete(uuid), "Failed to delete reactive entity!");
107        assert!(!reactive_entity_manager.has(uuid), "Reactive entity should have been deleted!");
108        let imported_reactive_entity = entity_instance_import_export_manager
109            .import(&path)
110            .await
111            .expect("Failed to import entity instance");
112        assert_eq!(uuid, imported_reactive_entity.id, "The imported reactive entity's id doesn't match");
113        assert!(reactive_entity_manager.has(uuid), "The reactive entity should have been registered during import");
