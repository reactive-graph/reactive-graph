12#[derive(Clone, Debug, Serialize, Deserialize, InputObject)]
13#[graphql(name = "InstanceAddress")]
14pub struct InstanceAddressDefinition {
15    /// The hostname.
16    pub hostname: String,
17
18    /// The port.
19    pub port: u16,
20
21    /// Secure endpoint.
22    pub secure: Option<bool>,
23
24    /// The user agent.
25    pub user_agent: Option<String>,
26
27    /// The relative URL of the GraphQL endpoint, by default "/graphql".
28    pub endpoint_graphql: Option<String>,
29
30    /// The relative URL of the dynamic graph endpoint, by default "/dynamic_graph".
31    pub endpoint_dynamic_graph: Option<String>,
32
33    /// The relative URL of the runtime endpoint, by default "/runtime/graphql".
34    pub endpoint_runtime: Option<String>,
35
36    /// The relative URL of the plugins endpoint, by default "/plugin/graphql".
37    pub endpoint_plugin: Option<String>,
38
39    /// The authentication token.
40    pub bearer: Option<String>,
41
42    /// Controls the use of certificate validation.
43    ///
44    /// Defaults to `false`.
45    ///
46    /// Warning
47    ///
48    /// You should think very carefully before using this method. If
49    /// invalid certificates are trusted, *any* certificate for *any* site
50    /// will be trusted for use. This includes expired certificates. This
51    /// introduces significant vulnerabilities, and should only be used
52    /// as a last resort.
53    danger_accept_invalid_certs: Option<bool>,
54
55    /// Controls the use of hostname verification.
56    ///
57    /// Defaults to `false`.
58    ///
59    /// Warning
60    ///
61    /// You should think very carefully before you use this method. If
62    /// hostname verification is not used, any valid certificate for any
63    /// site will be trusted for use from any other. This introduces a
64    /// significant vulnerability to man-in-the-middle attacks.
65    danger_accept_invalid_hostnames: Option<bool>,
66}
67
68impl From<InstanceAddressDefinition> for InstanceAddress {
69    fn from(address: InstanceAddressDefinition) -> Self {
70        InstanceAddress {
71            hostname: address.hostname,
72            port: address.port,
73            secure: address.secure.unwrap_or(false),
74            user_agent: address.user_agent.unwrap_or(DEFAULT_USER_AGENT.to_owned()),
75            endpoint_graphql: address.endpoint_graphql.unwrap_or(DEFAULT_ENDPOINT_GRAPHQL.to_owned()),
76            endpoint_dynamic_graph: address.endpoint_dynamic_graph.unwrap_or(DEFAULT_ENDPOINT_DYNAMIC_GRAPH.to_owned()),
77            endpoint_runtime: address.endpoint_runtime.unwrap_or(DEFAULT_ENDPOINT_RUNTIME.to_owned()),
78            endpoint_plugin: address.endpoint_plugin.unwrap_or(DEFAULT_ENDPOINT_PLUGIN.to_owned()),
79            bearer: address.bearer,
80            danger_accept_invalid_certs: address.danger_accept_invalid_certs,
81            danger_accept_invalid_hostnames: address.danger_accept_invalid_hostnames,
82        }
83    }
84}
