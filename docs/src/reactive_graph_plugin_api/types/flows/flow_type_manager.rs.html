35pub trait FlowTypeManager: Send + Sync {
36    /// Returns all flow types.
37    fn get_all(&self) -> FlowTypes;
38
39    /// Returns all flow types.
40    fn get_by_namespace(&self, namespace: &str) -> FlowTypes;
41
42    /// Returns true, if a flow type with the given name exists.
43    fn has(&self, ty: &FlowTypeId) -> bool;
44
45    /// Returns true, if a flow type with the given name exists.
46    fn has_by_type(&self, namespace: &str, name: &str) -> bool;
47
48    /// Returns the flow type with the given name or empty.
49    fn get(&self, ty: &FlowTypeId) -> Option<FlowType>;
50
51    /// Returns the flow type with the given name or empty.
52    fn get_by_type(&self, namespace: &str, name: &str) -> Option<FlowType>;
53
54    /// Returns all flow types whose names matches the given search string.
55    fn find_by_type_name(&self, search: &str) -> FlowTypes;
56
57    /// Returns the count of flow types.
58    fn count(&self) -> usize;
59
60    /// Returns the count of flow types of the given namespace.
61    fn count_by_namespace(&self, namespace: &str) -> usize;
62
63    /// Creates a new flow type.
64    #[allow(clippy::too_many_arguments)]
65    fn create(
66        &self,
67        ty: &FlowTypeId,
68        description: &str,
69        wrapper_entity_instance: EntityInstance,
70        entity_instances: EntityInstances,
71        relation_instances: RelationInstances,
72        variables: PropertyTypes,
73        extensions: Extensions,
74    ) -> Result<FlowType, FlowTypeCreationError>;
75
76    /// Updates the description of the given flow type.
77    fn update_description(&self, ty: &FlowTypeId, description: &str) -> Result<FlowType, FlowTypeUpdateError>;
78
79    /// Adds the given entity instance to the flow type with the given name.
80    fn add_entity_instance(&self, ty: &FlowTypeId, entity_instance: EntityInstance) -> Result<(), FlowTypeAddEntityInstanceError>;
81
82    /// Updates the entity instance with the given id of the flow type with the given name.
83    fn update_entity_instance(
84        &self,
85        ty: &FlowTypeId,
86        id: Uuid,
87        entity_instance: EntityInstance,
88    ) -> Result<(Uuid, EntityInstance), FlowTypeUpdateEntityInstanceError>;
89
90    /// Removes the entity instance with the given id from the flow type with the given name.
91    fn remove_entity_instance(&self, ty: &FlowTypeId, id: Uuid) -> Result<Option<(Uuid, EntityInstance)>, FlowTypeRemoveEntityInstanceError>;
92
93    /// Adds the given extension to the given flow type.
94    fn add_extension(&self, ty: &FlowTypeId, extension: Extension) -> Result<ExtensionTypeId, FlowTypeAddExtensionError>;
95
96    /// Updates the extension with the given type of the given flow type.
97    fn update_extension(&self, ty: &FlowTypeId, extension_ty: &ExtensionTypeId, extension: Extension) -> Result<Extension, FlowTypeUpdateExtensionError>;
98
99    /// Removes the extension with the given type from the given flow type.
100    fn remove_extension(&self, flow_ty: &FlowTypeId, extension_ty: &ExtensionTypeId) -> Result<Extension, FlowTypeRemoveExtensionError>;
101
102    /// Adds the given variable to the given flow type.
103    fn add_variable(&self, ty: &FlowTypeId, variable: PropertyType) -> Result<Variable, FlowTypeAddVariableError>;
104
105    /// Updates the variable with the given name of the flow type with the given name.
106    fn update_variable(&self, ty: &FlowTypeId, variable_name: &str, variable: PropertyType) -> Result<Variable, FlowTypeUpdateVariableError>;
107
108    /// Removes the variable with the given name from the flow type with the given name.
109    fn remove_variable(&self, ty: &FlowTypeId, variable_name: &str) -> Result<Variable, FlowTypeRemoveVariableError>;
110
111    /// Deletes the flow type with the given name.
112    fn delete(&self, ty: &FlowTypeId) -> Option<FlowType>;
113
114    /// Validates the flow type with the given name.
115    /// Tests that all entity types and relation types exists and are valid.
116    fn validate(&self, ty: &FlowTypeId) -> bool;
117}
