1use std::sync::Arc;
2
3use actix_web::get;
4use actix_web::web;
5use actix_web::HttpResponse;
6use mime::APPLICATION_JSON;
7
8use reactive_graph_type_system_api::EntityTypeManager;
9
10#[get("/types/entities")]
11pub async fn get_entity_types(entity_type_manager: web::Data<Arc<dyn EntityTypeManager + Send + Sync>>) -> HttpResponse {
12    HttpResponse::Ok()
13        .content_type(APPLICATION_JSON.to_string())
14        .json(entity_type_manager.get_all())
15}
16
17#[get("/types/entities/{namespace}/{type_name}")]
18pub async fn get_entity_type(path: web::Path<(String, String)>, entity_type_manager: web::Data<Arc<dyn EntityTypeManager + Send + Sync>>) -> HttpResponse {
19    let (namespace, type_name) = path.into_inner();
20    match entity_type_manager.get_by_type(&namespace, &type_name) {
21        Some(entity_type) => HttpResponse::Ok().content_type(APPLICATION_JSON.to_string()).json(&entity_type),
22        None => HttpResponse::NotFound()
23            .content_type(APPLICATION_JSON.to_string())
24            .body(format!("Entity Type {}__{} not found", namespace, type_name)),
25    }
26}
