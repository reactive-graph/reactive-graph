1use crate::DataType;

9#[derive(cynic::QueryFragment, Clone, Debug)]
10#[cynic(schema_path = "schema_graphql.graphql", schema_module = "crate::schema_graphql::schema")]
11pub struct PropertyType {
12    /// The name of the extension.
13    pub name: String,
14
15    /// Textual description of the extension.
16    pub description: String,
17
18    /// The extension as JSON representation.
19    pub data_type: DataType,
20
21    /// Specifies the type of socket - either input socket or output socket or none
22    pub socket_type: SocketType,
23
24    /// Specifies if the property is mutable.
25    pub mutability: Mutability,
26
27    /// Property specific extensions
28    pub extensions: Vec<Extension>,
29}
30
31impl From<PropertyType> for reactive_graph_graph::PropertyType {
32    fn from(property_type: PropertyType) -> Self {
33        reactive_graph_graph::PropertyType {
34            name: property_type.name,
35            description: property_type.description,
36            data_type: property_type.data_type.into(),
37            socket_type: property_type.socket_type.into(),
38            mutability: property_type.mutability.into(),
39            extensions: Extensions(property_type.extensions).into(),
40        }
41    }
42}
43
44#[derive(Clone, Debug)]
45pub struct PropertyTypes(pub Vec<PropertyType>);
46
47impl From<PropertyTypes> for reactive_graph_graph::PropertyTypes {
48    fn from(property_types: PropertyTypes) -> Self {
49        property_types.0.into_iter().map(|property_type| property_type.into()).collect()
50    }
51}
52
53impl fmt::Display for PropertyTypes {
54    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
55        writeln!(f)
