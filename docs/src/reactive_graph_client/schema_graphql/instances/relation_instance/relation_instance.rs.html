11#[derive(cynic::QueryFragment, Clone, Debug)]
12#[cynic(schema_path = "schema_graphql.graphql", schema_module = "crate::schema_graphql::schema")]
13pub struct RelationInstance {
14    inbound: EntityInstance,
15    #[cynic(rename = "type")]
16    ty: Option<RelationType>,
17    instance_id: String,
18    outbound: EntityInstance,
19    name: String,
20    description: String,
21    properties: Vec<PropertyInstance>,
22    components: Vec<Component>,
23}
24
25impl RelationInstance {
26    pub fn ty(&self) -> RelationTypeId {
27        self.ty
28            .clone()
29            .map(|relation_type| RelationTypeId::new_from_type(relation_type.namespace, relation_type.name))
30            .unwrap_or(RelationTypeId::new_from_type(String::new(), String::new()))
31    }
32
33    pub fn instance_ty(&self) -> RelationInstanceTypeId {
34        RelationInstanceTypeId::new_unique_for_instance_id(self.ty(), self.instance_id.clone())
35    }
36}

38impl From<RelationInstance> for reactive_graph_graph::RelationInstance {
39    fn from(relation_instance: RelationInstance) -> Self {
40        let ty = relation_instance.instance_ty();
41        let outbound_id = relation_instance.outbound.id.into();
42        let inbound_id = relation_instance.inbound.id.into();
43        let properties = PropertyInstances(relation_instance.properties).into();
44        let components = relation_instance
45            .components
46            .iter()
47            .map(|component| {
48                let ty: reactive_graph_graph::ComponentTypeId = component.clone().ty().into();
49                ty
50            })
51            .collect();
52        reactive_graph_graph::RelationInstance {
53            outbound_id,
54            ty,
55            inbound_id,
56            name: relation_instance.name.clone(),
57            description: relation_instance.description.clone(),
58            properties,
59            components,
60            extensions: Default::default(),
61        }
62    }
63}
