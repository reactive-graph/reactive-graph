44    pub async fn get_all_components(&self) -> Result<Option<Vec<Component>>, ReactiveGraphClientExecutionError> {
45        let components = self
46            .client
47            .client
48            .post(self.client.url_graphql())
49            .run_graphql(get_all_components_query())
50            .await
51            .map_err(ReactiveGraphClientExecutionError::FailedToSendRequest)?
52            .data
53            .map(|data| ComponentsVec(data.types.components))
54            .map(From::from);
55        Ok(components)
56    }

58    pub async fn get_component_by_type<C: Into<ComponentTypeId>>(&self, ty: C) -> Result<Option<Component>, ReactiveGraphClientExecutionError> {
59        let component = self
60            .client
61            .client
62            .post(self.client.url_graphql())
63            .run_graphql(get_component_by_type_query(&ty.into()))
64            .await
65            .map_err(ReactiveGraphClientExecutionError::FailedToSendRequest)?

70    }
