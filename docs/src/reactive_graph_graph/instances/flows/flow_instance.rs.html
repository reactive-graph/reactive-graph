38/// A flow instance is a container for entity instances and relation instances.
39///
40/// A flow instance is strictly associated with a wrapper entity instance. The properties
41/// of the wrapper entity instance are the properties of the flow.
42///
43/// Additionally, flows can be nested -  from the perspective of the outer flow
44/// the inner flow acts like an entity instance. The wrapper entity instance of
45/// the inner flow is the interface which can be accessed by the outer flow.
46///
47/// Entity instances and relation instances can be shared with multiple flows.
48///
49/// It's even possible to connect entity instances from different flows with relation
50/// instances.
51///
52#[derive(Clone, Debug, Eq, Hash, PartialEq, Serialize, Deserialize, JsonSchema, TypedBuilder)]
53#[schemars(deny_unknown_fields)]
54pub struct FlowInstance {
55    /// The id of the flow corresponds to the id of the wrapper entity instance
56    ///
57    /// This means the vector of entity instances must contain an instance with
58    /// the id of the flow.
59    pub id: Uuid,
60
61    /// The type definition of the entity type of the wrapper entity instance.
62    #[serde(flatten)]
63    #[builder(setter(into))]
64    pub ty: EntityTypeId,
65
66    /// The name of the flow instance.
67    #[serde(default = "String::new")]
68    #[builder(default, setter(into))]
69    pub name: String,
70
71    /// Textual description of the flow instance.
72    #[serde(default = "String::new")]
73    #[builder(default, setter(into))]
74    pub description: String,
75
76    /// The entity instances which are contained in this flow instance.
77    ///
78    /// It can't have a default because the wrapper entity instance must be
79    /// present in the list of entities.
80    #[serde(default = "EntityInstances::new", alias = "entities")]
81    #[builder(default, setter(into))]
82    pub entity_instances: EntityInstances,
83
84    /// The relation instances which are contained in this flow instance.
85    ///
86    /// By default, no relation instances are contained in this flow instance.
87    #[serde(default = "RelationInstances::new", alias = "relations")]
88    #[builder(default, setter(into))]
89    pub relation_instances: RelationInstances,
90}

92impl FlowInstance {
93    /// Constructs a new flow instance from the wrapper entity instance.
94    pub fn from_instance_with_name<S: Into<String>>(wrapper_entity_instance: EntityInstance, name: S) -> FlowInstance {
95        FlowInstance {
96            id: wrapper_entity_instance.id,
97            ty: wrapper_entity_instance.ty.clone(),
98            name: name.into(),
99            description: String::new(),
100            entity_instances: EntityInstances::new_with_instance(wrapper_entity_instance),
101            relation_instances: RelationInstances::new(),
102        }
103    }
104}

171impl FlowInstances {
172    pub fn new() -> Self {
173        FlowInstances(DashMap::new())
174    }
175
176    pub fn new_with_instance<E: Into<FlowInstance>>(flow_instance: E) -> Self {
177        let flow_instances = FlowInstances::new();
178        flow_instances.push(flow_instance.into());
179        flow_instances
180    }
181
182    pub fn push<E: Into<FlowInstance>>(&self, flow_instance: E) {
183        let flow_instance = flow_instance.into();
184        self.0.insert(flow_instance.id, flow_instance);
185    }
186
187    pub fn to_vec(&self) -> Vec<FlowInstance> {
188        let mut items: Vec<_> = self.iter().map(|item| item.value().clone()).collect();
189        items.sort();
190        items
191    }
