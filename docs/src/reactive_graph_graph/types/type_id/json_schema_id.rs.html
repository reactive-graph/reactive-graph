9pub const JSON_SCHEMA_ID_URI_PREFIX: &str = "https://schema.reactive-graph.io/schema/json";
10
11pub struct JsonSchemaId(TypeDefinition);
12
13impl JsonSchemaId {
14    pub fn new<TD: Into<TypeDefinition>>(type_definition: TD) -> Self {
15        Self(type_definition.into())
16    }
17}
18
19impl Display for JsonSchemaId {
20    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
21        write!(
22            f,
23            "{}/dynamic_graph/{}/{}/{}.schema.json",
24            JSON_SCHEMA_ID_URI_PREFIX,
25            self.0.type_id_type.full_name().to_lowercase(),
26            self.0.namespace,
27            self.0.type_name,
28        )
29    }
30}
31
32impl<T> From<&T> for JsonSchemaId

38    }
39}
40
41impl From<JsonSchemaId> for Value {

44    }
