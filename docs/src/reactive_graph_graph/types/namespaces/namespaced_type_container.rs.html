4use dashmap::DashMap;
5use dashmap::DashSet;
6use wildmatch::WildMatch;

8use crate::NamespacedTypeGetter;
9use crate::types::Namespaces;

11pub trait NamespacedTypeContainer
12where
13    Self: Sized + Deref<Target = DashMap<Self::TypeId, Self::Type>> + FromIterator<Self::Type>,
14    Self::TypeId: Clone + Eq + PartialEq + Hash,
15    Self::TypeIds: Clone + FromIterator<Self::TypeId>,
16    Self::Type: Clone + Hash + Ord + Sized + NamespacedTypeGetter,
17{
18    type TypeId;
19    type TypeIds;
20    type Type;
21
22    fn new() -> Self;
23
24    fn push<I: Into<Self::Type>>(&self, item_to_add: I);
25
26    fn type_ids(&self) -> Self::TypeIds {
27        self.iter().map(|item| item.key().clone()).collect()
28    }
29
30    fn types(&self) -> DashSet<Self::Type> {
31        self.iter().map(|item| item.value().clone()).collect()
32    }
33
34    fn to_vec(&self) -> Vec<Self::Type> {
35        let mut items: Vec<_> = self.iter().map(|item| item.value().clone()).collect();
36        items.sort();
37        items
38    }
39
40    fn namespaces(&self) -> Namespaces {
41        self.iter().map(|item| item.namespace()).collect()
42    }
43
44    fn get_by_namespace(&self, namespace: &str) -> Self {
45        self.iter()
46            .filter(|item| item.namespace() == namespace)
47            .map(|item| item.value().clone())
48            .collect()
49    }
50
51    fn get_types_by_namespace(&self, namespace: &str) -> Self::TypeIds {
52        self.iter()
53            .filter(|item| item.namespace() == namespace)
54            .map(|item| item.key().clone())
55            .collect()
56    }
57
58    fn find_by_type_name(&self, search: &str) -> Self {
59        let matcher = WildMatch::new(search);
60        self.iter()
61            .filter(|item| matcher.matches(item.type_name().as_str()))
62            .map(|item| item.value().clone())
63            .collect()
64    }
65
66    fn count_by_namespace(&self, namespace: &str) -> usize {
67        self.iter().filter(|item| item.namespace() == namespace).count()
68    }
69}
