11/// Addresses the component of an entity type.
12#[derive(Clone, Debug, PartialEq, Eq, Hash, Serialize, Deserialize, JsonSchema, TypedBuilder)]
13pub struct EntityComponentTypeId {
14    /// The entity type.
15    #[builder(setter(into))]
16    pub entity_ty: EntityTypeId,
17
18    /// The component type.
19    #[builder(setter(into))]
20    pub component_ty: ComponentTypeId,
21}
22
23impl EntityComponentTypeId {
24    pub fn new<E: Into<EntityTypeId>, C: Into<ComponentTypeId>>(entity_ty: E, component_ty: C) -> Self {
25        Self {
26            entity_ty: entity_ty.into(),
27            component_ty: component_ty.into(),
28        }
29    }
30}
31
32impl ComponentContainerGetter for EntityComponentTypeId {

36
37    fn component_ty(&self) -> ComponentTypeId {
