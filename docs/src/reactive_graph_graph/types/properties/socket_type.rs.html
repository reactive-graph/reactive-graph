12/// The socket type defines if the property acts as an input or output socket
13/// or is an hidden property
14#[derive(Clone, Debug, PartialEq, Eq, Hash, Copy, Serialize, Deserialize, JsonSchema)]
15#[serde(rename_all = "lowercase")]
16#[cfg_attr(any(test, feature = "test"), derive(RandGen))]
17pub enum SocketType {
18    /// The property doesn't act as input or output socket.
19    None,
20
21    /// The property acts as input socket and accepts incoming connections.
22    Input,
23
24    /// The property acts as output socket and accepts outgoing connections.
25    Output,
26}
27
28impl SocketType {
29    pub fn none() -> Self {
30        SocketType::None
31    }
32    pub fn input() -> Self {
33        SocketType::Input
34    }
35    pub fn output() -> Self {
36        SocketType::Output
37    }
38}
39
40impl From<&str> for SocketType {
41    fn from(value: &str) -> Self {
42        match value.to_lowercase().as_str() {
43            "none" => Self::None,
44            "input" => Self::Input,
45            "output" => Self::Output,
46            _ => Self::None,
47        }
48    }
49}
50
51impl Display for SocketType {
52    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
53        write!(f, "{:?}", self)
54    }
55}
