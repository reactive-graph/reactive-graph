4pub trait ComponentTypeIdContainer {
5    /// Returns true, if the type is composed with a component of the given type.
6    fn is_a(&self, ty: &ComponentTypeId) -> bool;
7
8    /// Returns true, if the type is composed with any of the given components.
9    fn is_any(&self, component_tys: &ComponentTypeIds) -> bool {
10        component_tys.iter().any(|ty| self.is_a(&ty))
11    }
12
13    /// Returns true, if the type is composed with every given components.
14    fn is_all(&self, component_tys: &ComponentTypeIds) -> bool {
15        component_tys.iter().all(|ty| self.is_a(&ty))
16    }
17
18    /// Adds the given component.
19    fn add_component<C: Into<ComponentTypeId>>(&self, ty: C) -> bool;
20
21    /// Adds the given components.
22    fn add_components<C: Into<ComponentTypeIds>>(&mut self, components_to_add: C);
23
24    /// Removes the component with the given type id.
25    fn remove_component(&self, ty: &ComponentTypeId) -> Option<ComponentTypeId>;
26
27    /// Removes the given components.
28    fn remove_components<C: Into<ComponentTypeIds>>(&mut self, components_to_remove: C);
29}
30
31pub trait NamespacedTypeComponentTypeIdContainer<T, AddComponentError, RemoveComponentError> {
32    /// Returns all types having the specified component.
33    fn get_by_having_component(&self, component_ty: &ComponentTypeId) -> Self;
34
35    /// Adds the component with the given component_name to the entity type with the given name.
36    fn add_component(&self, ty: &T, component_ty: &ComponentTypeId) -> Result<(), AddComponentError>;
37
38    /// Remove the component with the given component_name from the entity type with the given name.
39    fn remove_component(&self, ty: &T, component_ty: &ComponentTypeId) -> Result<ComponentTypeId, RemoveComponentError>;
40}
